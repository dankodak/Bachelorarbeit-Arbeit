  textformat=simple,% 
  justification=justified,% 
  singlelinecheck=off,% 
  font={% % Affects the whole figure caption
  },%
  labelfont={% % Affects only the label name and its number

    bf,%
    color=captionlabel,%
  },%
  textfont={% % Affects only the actual float caption
  },%
  subrefformat=simple,% 
  labelformat=simple,%
}

  position=above,%
  belowskip=2pt,%
}

  position=below,%
  belowskip=2pt,%
}

  belowskip=2pt,% 
  textfont={sl},
  labelformat=simple,
